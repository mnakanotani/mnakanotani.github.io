Charged Particle Motion¶

Equation of motion of particles can be written as follows $$m\frac{d\vec{v}}{dt}=q\left(\vec{E}+\frac{\vec{v}}{c}\times\vec{B}\right),\quad \frac{d\vec{r}}{dt}=\vec{v}.$$
The equations are solved by the Bunnemann-Boriss method.
I show a particle motion in X-type magnetic fields.

#for one particle
@jit
def EoM(vx0,vy0,vz0,h,nt):
    q=-1; m=1;
    ch=h*q/2/m;
    rx=1.;   ry=1;   rz=0
    vx=vx0; vy=vy0; vz=vz0
    
    mu=np.zeros(nt)
    r =np.zeros((3,nt))
    for k in range(nt):
        ex=0;ey=0;ez=0;
        bx=-ry;by=-rx;bz=0.0
        
        fx=ch*ex; fy=ch*ey; fz=ch*ez
        tx=ch*bx; ty=ch*by; tz=ch*bz
        tc=2/(1+tx**2+ty**2+tz**2)
        
        vmx=vx+fx 
        vmy=vy+fy
        vmz=vz+fz
        
        v0x=vmx+vmy*tz-vmz*ty
        v0y=vmy+vmz*tx-vmx*tz
        v0z=vmz+vmx*ty-vmy*tx
        
        vpx=vmx+tc*(v0y*tz-v0z*ty)
        vpy=vmy+tc*(v0z*tx-v0x*tz)
        vpz=vmz+tc*(v0x*ty-v0y*tx)
        
        vx=vpx+fx
        vy=vpy+fy
        vz=vpz+fz
        
        rx=rx+h*vx
        ry=ry+h*vy
        rz=rz+h*vz
        
        vpara=(vx*bx+vy*by+vz*bz)/mt.sqrt(bx**2+by**2+bz**2)
        vperp=mt.sqrt(vx**2+vy**2+vz**2-vpara**2)
        mu[k]=vperp**2/mt.sqrt(bx**2+by**2+bz**2)
        r[0,k]=rx
        r[1,k]=ry
        r[2,k]=rz
         
        if k%100==0:
            display.clear_output(True)
            #plt.figure()
            plt.subplots_adjust(wspace=0.4, hspace=0.6)
            plt.subplot(121)
            plt.plot(r[0,1:k],r[1,1:k]);plt.xlabel('x');plt.ylabel('y')
            plt.plot(rx,ry,'ro')
            n = 10
            X, Y = np.mgrid[-n:n, -n:n]
            U, V = -Y, -X
            plt.quiver(X, Y, U, V, 1, alpha=1.)
            plt.xlim(-10,10)
            plt.ylim(-10,10)
            
            plt.subplot(122)
            plt.plot(r[1,1:k],r[2,1:k]);plt.xlabel('y');plt.ylabel('z')
            plt.plot(ry,rz,'ro')
            #plt.xlim(-10,10)
            #plt.ylim(-10,10)

            plt.axes([0.125,-0.5,0.775,0.5])
            plt.plot(mu[1:k],'k');plt.xlabel('t');plt.ylabel(r'$\mu$')
            plt.plot(k,mu[k],'ro')
            plt.show()
