program sin_curve
  implicit none
  integer,parameter :: ng = 1000
  integer i,ig
  real(8) :: x(ng),pi

  open(17,file="sin_curve.d",status="replace",form="unformatted" ,access="stream")
  pi=3.1415926d0
  do i = 1,100
    write(17) (sin(2.d0*pi*ig/ng+i*0.1d0),ig=1,ng)
    !print *, x
  enddo
  
end program sin_curve
